WORKING WTH SCHEDULE IN PYTHON

# to install schedule
# RUN "python -m pip install schedule" in CMD

import schedule
import time

def job():
    print("I'm working...")

schedule.every(10).minutes.do(job)
schedule.every().hour.do(job)
schedule.every().day.at("10:30").do(job)
schedule.every().monday.do(job)
schedule.every().wednesday.at("13:15").do(job)
schedule.every().minute.at(":17").do(job)

while True:
    schedule.run_pending()
    time.sleep(1)

CONNECTING TO DATABASE USING SQLALCHEMY
#to set the database you first of set the 
		app.cofig[]="databasetype://database_name:database_password@host/database_you_created"

POSTGRES: app.cofig["SQLALCHEMY_DATABASE_URI"]="postgresql://beta:password@localhost:5432/car_db"

MYSQL:	app.config["SQLALCHEMY_DATABASE_URI"]="mysql+pymysql://beta:password@localhost/car_db"
Note: the 5432 is the port postgresql is running it is only applicable when setting postgresql
